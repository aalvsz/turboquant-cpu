def nth_fibonacci(n):
    # Initialize the Fibonacci number
    fib_n = [0] * (n + 1)

    # Compute the nth Fibonacci number
    fib_n[1] = 1

    # Compute the nth Fibonacci number
    for i in range(2, n + 1):
        fib_n[i] = (i - 1) * fib_n[i - 1]

    # Return the nth Fibonacci number
    return fib_n[n]
